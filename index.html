    <script>

      (() => {

        const reduceMotion = window.matchMedia?.("(prefers-reduced-motion: reduce)")?.matches;


        const haru = document.getElementById("haru");

        const audio = document.getElementById("bgm");


        // --- click vs dblclick 분리 (더블클릭 시 토글/재생이 2번 먹는 것 방지) ---

        let clickTimer = null;

        const CLICK_DELAY = 220;


        const doToggle = async () => {

          document.body.classList.toggle("dots-on");

          try {

            if (audio.paused) await audio.play();

            else audio.pause();

          } catch (e) {

            console.log(e);

          }

        };


        // Single click: toggle circles + toggle music

        haru.addEventListener("click", () => {

          if (clickTimer) return;

          clickTimer = setTimeout(() => {

            clickTimer = null;

            void doToggle();

          }, CLICK_DELAY);

        });


        // Double click: go to /about/

        haru.addEventListener("dblclick", (e) => {

          e.preventDefault();

          if (clickTimer) {

            clearTimeout(clickTimer);

            clickTimer = null;

          }

          window.location.href = "about/";

        });


        // Flash (skip if reduced motion)

        if (!reduceMotion) {

          let toggle = false;

          setInterval(() => {

            toggle = !toggle;

            const cls = toggle ? "flash-white" : "flash-black";

            document.body.classList.add(cls);

            setTimeout(() => document.body.classList.remove(cls), 50);

          }, 1000);

        }

      })();

    </script>
